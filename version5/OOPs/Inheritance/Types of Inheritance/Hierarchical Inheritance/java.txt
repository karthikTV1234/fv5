// Parent Class: Animal
class Animal {
    String name;

    // Constructor
    Animal(String name) {
        this.name = name;
    }

    // Common Method: Show name
    void showName() {
        System.out.println("Animal: " + name);
    }
}

// Child Class 1: Dog inherits from Animal
class Dog extends Animal {
    // Constructor
    Dog(String name) {
        super(name);
    }

    // Unique Method: Bark
    void bark() {
        System.out.println(name + " is barking üê∂");
    }
}

// Child Class 2: Cat inherits from Animal
class Cat extends Animal {
    // Constructor
    Cat(String name) {
        super(name);
    }

    // Unique Method: Meow
    void meow() {
        System.out.println(name + " is meowing üê±");
    }
}

// Main Class
public class Main {
    public static void main(String[] args) {
        // Create Dog object
        Dog dog = new Dog("Buddy");
        dog.showName();  // Inherited from Animal
        dog.bark();      // Dog's unique method

        System.out.println();  // Line break

        // Create Cat object
        Cat cat = new Cat("Whiskers");
        cat.showName(); // Inherited from Animal
        cat.meow();     // Cat's unique method
    }
}
